ajax.interceptors.request.use(function (response) {
    // 对响应数据做点什么
    console.log('相应拦截到了')
    return response
}, function (error) {
    if (error.response.status === 401) {
        window.alert('未登录， 即将跳转到登陆页面')
    } else if (error.response.status === 500) {
        window.alert('服务器正忙，请稍后')
    }
    return Promise.reject(error)
})
